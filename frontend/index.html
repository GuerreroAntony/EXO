🖥️ VPS

  • Ubuntu 24.04 LTS
  • Docker + Docker Compose
  • UFW: portas 22, 80, 443
  • CPU: 1.5 cores alocados
  • RAM: 3GB limite

🐳 Container

  • Imagem: openclaw/openclaw:latest
  • read_only + cap_drop ALL
  • no-new-privileges
  • Gateway: porta 18800
  • Voice: porta 3334

🌐 Dominios

  • callcenter.dominio.com → :18800
  • voice.dominio.com → :3334
  • TLS via Let's Encrypt
  • Caddy v2 reverse proxy

🔐 Seguranca

  • Nunca inventar informacoes
  • Nunca compartilhar dados entre clientes
  • Detectar prompt injection
  • Log de toda interacao
  • Opt-out imediato

  "agents": {
    "defaults": { "model": "claude-sonnet-4-6" },
    "list": [
      { "id": "recepcionista", "workspace": "/app/data/workspace-recepcionista", "model": "claude-sonnet-4-6" },
      { "id": "cobranca", "workspace": "/app/data/workspace-cobranca", "model": "claude-sonnet-4-6" },
      { "id": "sac", "workspace": "/app/data/workspace-sac", "model": "claude-haiku-4-5" },
      { "id": "agendamento", "workspace": "/app/data/workspace-agendamento", "model": "claude-haiku-4-5" }
    ]
  },
  "bindings": [
    { "agentId": "recepcionista", "match": { "channel": "voice", "direction": "inbound" } },
    { "agentId": "sac", "match": { "channel": "whatsapp" } }
  ]
}
